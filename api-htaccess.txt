PassengerEnabled on
PassengerAppType node
PassengerStartupFile server.js
PassengerAppRoot /home/falolega/eastafricanip.com/api
PassengerAppLogFile /home/falolega/logs/passenger.log

# Force all requests through Passenger
PassengerResolveSymlinksInDocumentRoot on

# Set environment variables
SetEnv NODE_ENV production
SetEnv PORT 3001

# Disable directory browsing
Options -Indexes

# Block all source files EXCEPT server.js (needed by Passenger)
<Files "server.js">
    Order allow,deny
    Allow from all
</Files>

<FilesMatch "\.(ts|tsx|jsx|json|md)$">
    Order allow,deny
    Deny from all
</FilesMatch>

# Block individual route files that shouldn't be accessed directly
<FilesMatch "^(auth|cases|clients|deadlines|financials|documents|system|jurisdictions|notes|oppositions|fees|notifications|forms|db|middleware|utils)\.js$">
    Order allow,deny
    Deny from all
</FilesMatch>

RewriteEngine On
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule . / [L,QSA]
